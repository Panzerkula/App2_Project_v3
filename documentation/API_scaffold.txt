//----------------USER ACCOUNTS---------------------

Create user:
    POST /auth/signup
        IF username !taken, email=valid and acceptTos=true{
            CREATE new user
        }else{
            RETURN error
        }


Delete own user:
    DELETE /auth/me

Edit own user:
    PUT /auth/me

Log in:
    POST /auth/login
        IF username, password combo exists{
            RETURN user logged in
        }ELSE IF username, password combo exists and is admin verified{
            RETURN admin logged in
        }ELSE{
            RETURN invalid username and/or password
        }


Current session:
    GET /auth/me
        RETURN session user OR null

//----------------CORE APP-----------------------

Create game:
    POST /games
        IF user is authenticated{

            CREATE new game
            owner = current user
            players = submitted players

            SAVE game
            RETURN game id
        }


List games user:
    GET /games/user
        REQUIRE user is authenticated

        FIND games WHERE user is a player
        RETURN game summaries


Game details user:
    GET /games/user/games/:gameId
        REQUIRE user is authenticated
        REQUIRE user is player in game

        RETURN full game data


Edit game details user:
    PUT /games/user/games/:gameId
        REQUIRE user is authenticated
        REQUIRE user is owner of game

        RETURN new full game data

    GET /games

Admin game details:

List users (for admins):
    GET /auth/users
        REQUIRE user is admin verified

        RETURN all users


Edit specific user:
    PUT /auth/users/:userId
        REQUIRE user is admin verified

        RETURN user


Delete specific user: (Might switch endpoint to /users/:userId eventually)

        REQUIRE user is admin verified
