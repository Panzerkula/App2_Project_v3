Create user:
    POST /auth/signup
        IF username !taken, email=valid and acceptTos=true{
            CREATE new user
        }else{
            RETURN error
        }


Delete user:
    DELETE /auth/signup/me


Log in:
    POST /auth/login
        IF username, password combo exists{
            RETURN user logged in
        }ELSE IF username, password combo exists and is admin verified{
            RETURN admin logged in
        }ELSE{
            RETURN invalid username and/or password
        }


Current session:
    GET /auth/me
        RETURN session user OR null


Create game:
    POST /games
        IF user is authenticated{

            CREATE new game
            owner = current user
            players = submitted players

            SAVE game
            RETURN game id
        }


List games user:
    GET /games/user
        REQUIRE user is authenticated

        FIND games WHERE user is a player
        RETURN game summaries


List game details:
    GET /games/user/:gameId
        REQUIRE user is authenticated
        REQUIRE user is player in game

        RETURN full game data


List games admin:
    GET /games
        REQUIRE user is admin verified
        
        RETURN all games


List admin game details:
    GET /games/:gameId
        REQUIRE user is admin verified
        
        RETURN full game data


List users (for admins):
    GET /auth/users
        REQUIRE user is admin verified

        RETURN all users
