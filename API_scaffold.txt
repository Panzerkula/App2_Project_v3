Current session:
    GET /auth/me
        RETURN session user OR null


Create game:
    POST /games
        IF user is authenticated{

            CREATE new game
            owner = current user
            players = submitted players

            SAVE game
            RETURN game id
        }


List games:
    GET /games
        REQUIRE user is authenticated

        FIND games WHERE user is a player
        RETURN game summaries


List game details:
    GET /games/:gameId
        REQUIRE user is authenticated
        REQUIRE user is player in game

        RETURN full game data
